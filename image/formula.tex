牛顿算法
x_{i+1}=x_{i}-\frac{f(x_{i})}{{f(x_{i})}'};i=0,1,2,...

m重根改进牛顿算法
x_{i+1}=x_{i}-\frac{mf(x_{i})}{{f(x_{i})}'};\ m>1;i=0,1,2,...


牛顿算法推导
\begin{aligned}
y-f(x_{0})&={f(x_{0})}'(x-x_{0})\\
0-f(x_{0})&={f(x_{0})}'(x-x_{0})\\
x&=x_{0}-\frac{f(x_{0})}{{f(x_{0})}'}\\
set\ x_{1}&=x_{0}-\frac{f(x_{0})}{{f(x_{0})}'}
\end{aligned}

使用牛顿法开方公式
\begin{aligned}
f(x)&=x^{2}-a\\
x_{i+1}&=x_{i}-\frac{f(x_{i})}{{f(x_{i})}'}\\
&=x_{i}-\frac{x_{i}^{2}-a}{{2x_{i}}}\\
&=\frac{x_{i}^{2}+a}{{2x_{i}}}\\
&=\frac{x_{i}+\frac{a}{x_{i}}}{2}
\end{aligned}

牛顿二次收敛定理
\begin{aligned}
M&=\lim_{n \to \infty }
\frac{e_{i+1}}{e_{i}^{2}}<\infty \\
if\ f(r)&=0;\ f'(r)\neq 0\\
\lim_{n \to \infty }
\frac{e_{i+1}}{e_{i}^{2}}&=M=\frac{f''(r)}{2f'(r)}\\
\end{aligned}

\begin{aligned}
(b-A\overline{x})\perp \{Ax|x\in R^n\}\\

x^TA^T(b-A\overline{x})=0\\

A^TA\overline{x}=A^Tb
